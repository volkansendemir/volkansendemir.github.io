Sudoku Generator/Solver

Solved on Sat, 10 Oct 2020
Uploaded on Sat, 19 Dec 2020

This project generates a solvable Sudoku and can solve a generated Sudoku.

The generation process consists of two stages:
1) constructing a filled, completed Sudoku
2) removing digits from the filled Sudoku one-by-one, making sure it is still solvable

The solving stage iterates over empty fields and fills them if only a single value is possible.
It can solve self-generated Sudoku's but may fail if a more complex Sudoku is fed as input.

Feel free to download the file and test it out!

Generation


Sub generateSudoku()
    randomFill
    randomRemove
End Sub

Filling the Grid


Sub randomFill():
    Application.StatusBar = "Creating random grid"
    createSudokuLayout
    fillRandomGrid
    Do While gridBroken
        If rowIncomplete Then filterGrid
        transposeGrid
        hasQuickFix
    Loop
    Application.StatusBar = "Random grid created"
End Sub

Creating the Layout


Private Sub createSudokuLayout()
    Cells.Interior.Color = RGB(255, 255, 255)
    cw = 6
    For i = 1 To 9
        ActiveSheet.Columns(i).ColumnWidth = cw
    Next i
    For i = 1 To 9
        ActiveSheet.Rows(i).RowHeight = 5.95 * cw
    Next i
    For i = 7 To 12
        ActiveSheet.Range(ActiveSheet.Cells(1, 1), ActiveSheet.Cells(9, 9)).Borders(i).LineStyle = xlContinuous
    Next i
    For i = 7 To 10
        For j = 1 To 9 Step 3
            For k = 1 To 9 Step 3
                ActiveSheet.Range(ActiveSheet.Cells(j, k), ActiveSheet.Cells(j + 2, k + 2)).Borders(i).Weight = xlThick
            Next k
        Next j

    ActiveSheet.Range(ActiveSheet.Cells(1, 1), ActiveSheet.Cells(9, 9)).Font.Color = RGB(0, 0, 0)
End Sub

Filling at random


Private Sub fillRandomGrid()
    For i = 1 To 3
        For j = 1 To 3
            iArray = randomi
            startx = ((i - 1) * 3) + 1
            starty = ((j - 1) * 3) + 1
            endx = i * 3
            endy = j * 3
            Range(Cells(startx, starty), Cells(endx, endy)).Value = iArray
        Next j
    Next i
End Sub

Private Function randomi() As Variant
    Dim newRand As Integer, canAdd As Boolean
    Dim iArray(1 To 3, 1 To 3) As Integer
    For i = 1 To 3
        For j = 1 To 3
            newRand = 1 + Int(9 * Rnd())
            canAdd = True
            For k = 1 To 3
                For l = 1 To 3
                    If iArray(k, l) = newRand Then canAdd = False
                Next l
            Next k
            If canAdd Then
                iArray(i, j) = newRand
            Else
                j = j - 1
            End If
        Next j
    Next i
    randomi = iArray
End Function

Checking for errors


Private Function gridBroken() As Boolean
    For i = 1 To 3
        For j = 1 To 3
            ReDim hasEach(1 To 9) As Boolean
            For k = 1 To 3
                For l = 1 To 3
                    hasEach(Cells((i - 1) * 3 + k, (j - 1) * 3 + l).Value) = True
                Next l
            Next k
            If Not Application.WorksheetFunction.And(hasEach) Then
                gridBroken = True
                Exit Function
            End If
        Next j
    Next i
    For i = 1 To 9
        ReDim hasEach(1 To 9) As Boolean
        For j = 1 To 9
            hasEach(Cells(i, j).Value) = True
        Next j
        If Not Application.WorksheetFunction.And(hasEach) Then
            gridBroken = True
            Exit Function
        End If
    Next i
    For i = 1 To 9
        ReDim hasEach(1 To 9) As Boolean
        For j = 1 To 9
            hasEach(Cells(j, i).Value) = True
        Next j
        If Not Application.WorksheetFunction.And(hasEach) Then
            gridBroken = True
            Exit Function
        End If
    Next i
    gridBroken = False
End Function

Private Function rowIncomplete() As Boolean
    For i = 1 To 9
        rSum = 0
        For j = 1 To 9
            rSum = rSum + Cells(i, j).Value
        Next j
        If rSum <> 45 Then
            rowIncomplete = True
            Exit Function
        End If
    Next i
    rowIncomplete = False
End Function

Filtering for rules


Private Sub filterGrid()
    Dim cRange As Range
    Dim bc As Integer
    For bc = 1 To 8
        iArray = rowCheck(bc)
        For i = bc To 9
            Set cRange = Cells(bc, i)
            If iArray(cRange.Value) > 1 Then
                hasChange = getPosbsRow(cRange, iArray)
                If hasChange Then i = bc
            End If
        Next i
        iArray = columnCheck(bc)
        For i = bc + 1 To 9
            Set cRange = Cells(i, bc)
            If iArray(cRange.Value) > 1 Then
                hasChange = getPosbsColumn(cRange, iArray)
                If hasChange Then i = bc + 1
            End If
        Next i
    Next bc
End Sub



Private Function rowCheck(rowIndex As Integer) As Variant
    Dim rowCounts(1 To 9) As Integer
    For i = 1 To 9
         rowCounts(Cells(rowIndex, i).Value) = rowCounts(Cells(rowIndex, i).Value) + 1
    Next i
    rowCheck = rowCounts
End Function

Private Function columnCheck(columnIndex As Integer) As Variant
    Dim columnCounts(1 To 9) As Integer
    For i = 1 To 9
        columnCounts(Cells(i, columnIndex).Value) = columnCounts(Cells(i, columnIndex).Value) + 1
    Next i
    columnCheck = columnCounts
End Function

Finding Alternatives


Private Function getPosbsRow(cRange As Range, iArray As Variant) As Boolean
    
    rIndex = cRange.Row
    cIndex = cRange.Column
    
    Dim quadRowMin As Integer, quadColMin As Integer, quadRowMax As Integer, quadColMax As Integer
    
    quadRowMin = rIndex + 1
    quadColMin = Application.WorksheetFunction.Max(rIndex + 1, Int((cIndex - 1) / 3) * 3 + 1)
    quadRowMax = Int((rIndex - 1) / 3) * 3 + 3
    quadColMax = Int((cIndex - 1) / 3) * 3 + 3
    
    For i = 1 To 9
        If iArray(i) = 0 Then
            For j = quadRowMin To quadRowMax
                For k = quadColMin To quadColMax
                    If Cells(j, k).Value = i Then
                        Cells(j, k).Value = cRange.Value
                        cRange.Value = i
                        iArray(i) = 1
                        iArray(Cells(j, k).Value) = iArray(Cells(j, k).Value) - 1
                        getPosbsRow = True
                        Exit Function
                    End If
                Next k
            Next j
        End If
    Next i
    getPosbsRow = False
End Function

Private Function getPosbsColumn(cRange As Range, iArray As Variant) As Boolean
    
    rIndex = cRange.Row
    cIndex = cRange.Column
    
    Dim quadRowMin As Integer, quadColMin As Integer, quadRowMax As Integer, quadColMax As Integer
    
    quadRowMin = Application.WorksheetFunction.Max(cIndex + 1, Int((rIndex - 1) / 3) * 3 + 1)
    quadColMin = cIndex + 1
    quadRowMax = Int((rIndex - 1) / 3) * 3 + 3
    quadColMax = Int((cIndex - 1) / 3) * 3 + 3
    
    For i = 1 To 9
        If iArray(i) = 0 Then
            For j = quadRowMin To quadRowMax
                For k = quadColMin To quadColMax
                    If Cells(j, k).Value = i Then
                        canProceed = True
                        For r = 1 To cIndex
                            If Cells(rIndex, r).Value = i Then canProceed = False
                        Next r
                        If canProceed Then
                            Cells(j, k).Value = cRange.Value
                            cRange.Value = i
                            iArray(i) = 1
                            iArray(Cells(j, k).Value) = iArray(Cells(j, k).Value) - 1
                            getPosbsColumn = True
                            Exit Function
                        End If
                    End If
                Next k
            Next j
        End If
    Next i
    getPosbsColumn = False
End Function

Reordering with rules


Private Sub transposeGrid()
    Dim mArray(1 To 9, 1 To 9) As Integer
    For i = 1 To 9
        For j = 1 To 9
            mArray(i, j) = Cells(i, j).Value
        Next j
    Next i
    
    For i = 9 To 1 Step -1
        ReDim rArray(1 To 9) As Integer
        For j = 1 To 9
            rArray(j) = mArray(9 - i + 1, j)
        Next j
        Range(Cells(1, i), Cells(9, i)).Value = Application.WorksheetFunction.Transpose(rArray)
    Next i
    
End Sub

Private Sub hasQuickFix()
    
    Dim i As Integer, j As Integer, k As Integer
    Dim quadrantFail As Integer, qfOne As Integer, qfTwo As Integer, qfThree As Integer
    
    For i = 1 To 3
        
        quadrantFail = 0
        qfOne = 0
        qfTwo = 0
        qfThree = 0
        For j = 1 To 3
            
            ReDim eachCount(1 To 9) As Integer
            For k = 1 To 9
                eachCount(Cells((((i - 1) * 3) + j), k).Value) = 1
            Next k
            If Application.WorksheetFunction.Sum(eachCount) <> 9 Then
                quadrantFail = quadrantFail + 1
                If qfOne = 0 Then
                    qfOne = ((i - 1) * 3) + j
                ElseIf qfTwo = 0 Then
                    qfTwo = ((i - 1) * 3) + j
                Else
                    qfThree = ((i - 1) * 3) + j
                End If
            End If
            
        Next j
        
        If quadrantFail = 2 Then
            fixQuadRow qfOne, qfTwo
        ElseIf quadrantFail = 3 Then
            fixQuadRow qfOne, qfTwo, qfThree
        End If
        
        quadrantFail = 0
        qfOne = 0
        qfTwo = 0
        qfThree = 0
        For j = 1 To 3
            
            ReDim eachCount(1 To 9) As Integer
            For k = 1 To 9
                eachCount(Cells(k, (((i - 1) * 3) + j)).Value) = 1
            Next k
            If Application.WorksheetFunction.Sum(eachCount) <> 9 Then
                quadrantFail = quadrantFail + 1
                If qfOne = 0 Then
                    qfOne = ((i - 1) * 3) + j
                ElseIf qfTwo = 0 Then
                    qfTwo = ((i - 1) * 3) + j
                Else
                    qfThree = ((i - 1) * 3) + j
                End If
            End If
            
        Next j
        
        If quadrantFail = 2 Then
            fixQuadCol qfOne, qfTwo
        ElseIf quadrantFail = 3 Then
            fixQuadCol qfOne, qfTwo, qfThree
        End If
        
    Next i
    
End Sub



Private Sub fixQuadRow(qfOne As Integer, qfTwo As Integer, Optional qfThree As Integer = 0)
    
    Dim i As Integer, j As Integer
    Dim lastSeen As Integer
    ReDim wasChanged(1 To 9) As Boolean
    limt = 0
    Do While Not rowFixed(qfOne) And limt < 10
        limt = limt + 1
        
        For i = 1 To 9
            lastSeen = 0
            For j = 1 To 9
                If Cells(qfOne, j).Value = i Then
                    If lastSeen = 0 Then
                        lastSeen = j
                    Else
                        If qfThree <> 0 Then
                            If Rnd() > 0.66666 Then
                                Cells(qfOne, lastSeen).Value = Cells(qfTwo, lastSeen).Value
                                Cells(qfTwo, lastSeen).Value = i
                                Exit For
                            ElseIf Rnd() > 0.33333 Then
                                Cells(qfOne, j).Value = Cells(qfTwo, j).Value
                                Cells(qfTwo, j).Value = i
                                Exit For
                            Else
                                Cells(qfOne, lastSeen).Value = Cells(qfThree, lastSeen).Value
                                Cells(qfThree, lastSeen).Value = i
                                Exit For
                            End If
                        ElseIf Not wasChanged(lastSeen) Then
                            Cells(qfOne, lastSeen).Value = Cells(qfTwo, lastSeen).Value
                            Cells(qfTwo, lastSeen).Value = i
                            wasChanged(lastSeen) = True
                            Exit For
                        Else
                            Cells(qfOne, j).Value = Cells(qfTwo, j).Value
                            Cells(qfTwo, j).Value = i
                            wasChanged(j) = True
                            Exit For
                        End If
                    End If
                End If
            Next j
        Next i
        
    Loop
    
End Sub

Private Function rowFixed(rIndex As Integer) As Boolean
    
    Dim eachDigit(1 To 9) As Integer, i As Integer
    For i = 1 To 9
        eachDigit(Cells(rIndex, i)) = 1
    Next i
    
    rowFixed = Application.WorksheetFunction.Sum(eachDigit) = 9
    
End Function



Private Sub fixQuadCol(qfOne As Integer, qfTwo As Integer, Optional qfThree As Integer = 0)
    
    Dim i As Integer, j As Integer
    Dim lastSeen As Integer
    ReDim wasChanged(1 To 9) As Boolean
    limt = 0
    Do While Not colFixed(qfOne) And limt < 10
        limt = limt + 1
        
        For i = 1 To 9 'for each digit
            lastSeen = 0
            For j = 1 To 9 'for each column
                If Cells(j, qfOne).Value = i Then
                    If lastSeen = 0 Then
                        lastSeen = j
                    Else
                        If qfThree <> 0 Then
                            If Rnd() > 0.66666 Then
                                Cells(lastSeen, qfOne).Value = Cells(lastSeen, qfTwo).Value
                                Cells(lastSeen, qfTwo).Value = i
                                Exit For
                            ElseIf Rnd() > 0.33333 Then
                                Cells(lastSeen, qfOne).Value = Cells(lastSeen, qfTwo).Value
                                Cells(lastSeen, qfTwo).Value = i
                                Exit For
                            Else
                                Cells(lastSeen, qfOne).Value = Cells(lastSeen, qfThree).Value
                                Cells(lastSeen, qfThree).Value = i
                                Exit For
                            End If
                        ElseIf Not wasChanged(lastSeen) Then
                            Cells(lastSeen, qfOne).Value = Cells(lastSeen, qfTwo).Value
                            Cells(lastSeen, qfTwo).Value = i
                            wasChanged(lastSeen) = True
                            Exit For
                        Else
                            Cells(j, qfOne).Value = Cells(j, qfTwo).Value
                            Cells(j, qfTwo).Value = i
                            wasChanged(j) = True
                            Exit For
                        End If
                    End If
                End If
            Next j
        Next i
        
    Loop
    
End Sub

Private Function colFixed(cIndex As Integer) As Boolean
    
    Dim eachDigit(1 To 9) As Integer, i As Integer
    For i = 1 To 9
        eachDigit(Cells(i, cIndex)) = 1
    Next i
    
    colFixed = Application.WorksheetFunction.Sum(eachDigit) = 9
    
End Function

Removing with Rules


Sub randomRemove():
    
    Application.StatusBar = "Removing at random"
    
    'Range(Cells(1, 1), Cells(9, 9)).Copy Range(Cells(1, 17), Cells(9, 25))
    
    Dim cRange As Range
    Dim rIndex As Integer, cIndex As Integer
    Dim deleteCount As Integer
    deleteCount = 0
    
    limt = 0
    Do While deleteCount < 39 And limt < 200
        limt = limt + 1
        
        rIndex = Int(Rnd * 9) + 1
        cIndex = Int(Rnd * 9) + 1
        
        Set cRange = Cells(rIndex, cIndex)
        If cRange.Value <> "" Then
            
            canBe = getPossibilities(cRange)
            
            cPos = Application.WorksheetFunction.Sum(canBe)
            If cPos = 1 Then
                cRange.ClearContents
                deleteCount = deleteCount + 1
            End If
            
        End If
    Loop
    
    For i = 1 To 9
        For j = 1 To 9
            If Cells(i, j).Value <> "" Then
                Cells(i, j).Font.Bold = True
                Cells(i, j).Font.Color = RGB(192, 0, 0)
            End If
        Next j
    Next i
    
    Application.StatusBar = "Ready"
End Sub



Private Function getPossibilities(cRange As Range) As Variant
    
    Dim canBe(1 To 9) As Integer
    
    For i = 1 To 9
        canBe(i) = 1
    Next i
    
    For i = 1 To 9
        
        If i <> cRange.Column And Cells(cRange.Row, i).Value <> "" Then canBe(Cells(cRange.Row, i).Value) = 0
        If i <> cRange.Row And Cells(i, cRange.Column).Value <> "" Then canBe(Cells(i, cRange.Column).Value) = 0
        
        qRow = Int((i - 1) / 3) + 1 + Int((cRange.Row - 1) / 3) * 3
        qCol = ((i - 1) Mod 3) + 1 + Int((cRange.Column - 1) / 3) * 3
        
        If (qRow <> cRange.Row Or qCol <> cRange.Column) And Cells(qRow, qCol).Value <> "" Then canBe(Cells(qRow, qCol).Value) = 0
        
    Next i
    
    getPossibilities = canBe
    
End Function

Solving the Puzzle


Sub solveSudoku():
    
    Dim cRange As Range
    Dim needsContinue As Boolean
    needsContinue = True
    
    Do While needsContinue
        needsContinue = False
        For i = 1 To 9
            For j = 1 To 9
                Set cRange = Cells(i, j)
                If cRange = "" Then
                    needsContinue = True
                    canBe = getPossibilities(cRange)
                    If Application.WorksheetFunction.Sum(canBe) = 1 Then
                        For k = 1 To 9
                            If canBe(k) = 1 Then cRange.Value = k
                            cRange.Select
                        Next k
                    End If
                End If
            Next j
        Next i
    Loop
    
    Cells(5, 13).Select
    
End Sub

End Function
