Starting in the top left corner of a 2×2 grid, and only being able to move to the right and down, there are exactly 6 routes to the bottom right corner.

How many such routes are there through a 20×20 grid?

Brute-force solution


def brute():
    grid = {}
    for outer in range(1, (lim_outer + 1)):
        grid[str(outer)] = {}

    a = 1
    for outer in range(a, (lim_outer + 1)):
        for inner in range(outer, (lim_inner + 1)):
            if inner == 1:
                grid[str(outer)][str(inner)] = 2
            elif outer == inner:
                val = grid[str(outer - 1)][str(inner)]
                for i in range(outer - 1):
                    val = val + grid[str(outer - 1 - i)][str(inner - 1)]
                grid[str(outer)][str(inner)] = val + 1
            else:
                val = 0
                for i in range(outer):
                    val = val + grid[str(outer - i)][str(inner - 1)]
                grid[str(outer)][str(inner)] = val + 1
    print(grid[str(lim_outer)][str(lim_inner)])

Execution


lim_outer = 20
lim_inner = 20
brute()
