Number letter counts

Solved on Mon, 19 Oct 2020
Uploaded on Thu, 17 Dec 2020

If the numbers 1 to 5 are written out in words: one, two, three, four, five, then there are 3 + 3 + 5 + 4 + 4 = 19 letters used in total.

If all the numbers from 1 to 1000 (one thousand) inclusive were written out in words, how many letters would be used?

NOTE: Do not count spaces or hyphens. For example, 342 (three hundred and forty-two) contains 23 letters and 115 (one hundred and fifteen) contains 20 letters. The use of "and" when writing out numbers is in compliance with British usage.

Brute-force solution


def brute_force():
    total_length = 0
    for i in range(1, lim_up + 1):
        strnum = ""
        numstr = str(i)
        if len(numstr) > 3:
            strnum = "onethousand"
        elif len(numstr) > 2:
            strnum = strnum + digits[numstr[0]]
            if numstr[1:] == "00":
                strnum = strnum + "hundred"
            else:
                strnum = strnum + "hundredand"
                if numstr[2:] == "0":
                    strnum = strnum + tens[numstr[1]]
                elif numstr[1] == "0":
                    strnum = strnum + digits[numstr[2]]
                elif numstr[1] == "1":
                    if numstr[2] == "1":
                        strnum = strnum + "eleven"
                    elif numstr[2] == "2":
                        strnum = strnum + "twelve"
                    elif numstr[2] == "3":
                        strnum = strnum + "thirteen"
                    elif numstr[2] == "5":
                        strnum = strnum + "fifteen"
                    elif numstr[2] == "8":
                        strnum = strnum + "eighteen"
                    else:
                        strnum = strnum + digits[numstr[2]]
                        strnum = strnum + "teen"
                else:
                    strnum = strnum + tens[numstr[1]]
                    strnum = strnum + digits[numstr[2]]
        elif len(numstr) > 1:
            if numstr[1] == "0":
                strnum = strnum + tens[numstr[0]]
            elif numstr[0] == "1":
                if numstr[1] == "1":
                    strnum = strnum + "eleven"
                elif numstr[1] == "2":
                    strnum = strnum + "twelve"
                elif numstr[1] == "3":
                    strnum = strnum + "thirteen"
                elif numstr[1] == "5":
                    strnum = strnum + "fifteen"
                elif numstr[1] == "8":
                    strnum = strnum + "eighteen"
                else:
                    strnum = strnum + digits[numstr[1]]
                    strnum = strnum + "teen"
            else:
                strnum = strnum + tens[numstr[0]]
                strnum = strnum + digits[numstr[1]]
        else:
            strnum = strnum + digits[numstr[0]]
        total_length = total_length + len(strnum)

    print(total_length)
