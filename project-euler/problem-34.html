145 is a curious number, as 1! + 4! + 5! = 1 + 24 + 120 = 145.

Find the sum of all numbers which are equal to the sum of the factorial of their digits.

Note: As 1! = 1 and 2! = 2 are not sums they are not included.

Brute-force solution


def brute():

    for i in range(3, lim_up):
        isum = 0

            fact = factorial(int(s))
            isum = isum + fact
        if isum == i:
            print(i)
            sumi = sumi + i

from matplotlib import pyplot as plt


def factorial(num):
    if num == 0: return 1
    fact = num
    for i in range(1, num):
        fact = fact * i
    return fact


def visualize_limit():
    nine_factorial = factorial(9)
    x = [[], [], []]
    y = [[], [], []]
    for i in range(nine_factorial * 42):
        sumi = 0
        all_nine = True
        for s in str(i):
            sumi = sumi + factorial(int(s))
            if s != "9": all_nine = False
        x[0].append(i)
        y[0].append(sumi)
        if all_nine:
            x[1].append(i)
            y[1].append(i)
        if i % nine_factorial == 0:
            x[2].append(i)
            y[2].append(i)
    fig, ax = plt.subplots()
    ax.scatter(x[0], y[0], s=1, c="grey")
    ax.scatter(x[2], y[2], s=3, c="red")
    ax.scatter(x[1], y[1], s=4, c="blue")
    plt.axvline(x=nine_factorial * 7, lw=1, ls="--")
    ax.ticklabel_format(useOffset=False, style="plain")
    plt.show()
