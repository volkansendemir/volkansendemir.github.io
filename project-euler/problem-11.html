What is the greatest product of four adjacent numbers in the same direction (up, down, left, right, or diagonally) in the 20×20 grid?

Strategic solution


def brute():
    max_prod = 0
    for i in range(len(grid) - factor):
        for j in range(len(grid) - factor):
            temp_grid = []
            for k in range(factor):
                temp_row = []
                for l in range(factor):
                    temp_row.append(int(grid[i + k][j + l]))
                temp_grid.append(temp_row)
            dia_max_1 = 1
            dia_max_2 = 1
            for k in range(factor):
                row_prod = 1
                col_prod = 1
                for l in range(factor):
                    row_prod = row_prod * temp_grid[k][l]
                    col_prod = col_prod * temp_grid[l][k]
                if row_prod > max_prod: max_prod = row_prod
                if col_prod > max_prod: max_prod = col_prod
                dia_max_1 = dia_max_1 * temp_grid[k][k]
                dia_max_2 = dia_max_2 * temp_grid[factor - k - 1][k]
            if dia_max_1 > max_prod: max_prod = dia_max_1
            if dia_max_2 > max_prod: max_prod = dia_max_2
    print(max_prod)
