In the United Kingdom the currency is made up of pound (£) and pence (p). There are eight coins in general circulation:

1p, 2p, 5p, 10p, 20p, 50p, £1 (100p), and £2 (200p).

It is possible to make £2 in the following way:

(1 * £1) + (1 * 50p) + (2 * 20p) + (1 * 5p) + (1 * 2p) + (3 * 1p)

How many different ways can £2 be made using any number of coins?

Brute-force solution


def check(mults):
    return lim_up == sum([mults[i] * coins[i] for i in range(len(mults))])


def brute():
    s = 1
    reps = [int(lim_up / coins[i]) + 1 for i in range(len(coins) - 1)]
    mults = [0 for coin in coins[:-1]]
    for i in range(reps[0]):
        mults[0] = i
        if sum(mults[u] * coins[u] for u in range(1)) > lim_up: break
        for j in range(reps[1]):
            mults[1] = j
            if sum(mults[u] * coins[u] for u in range(2)) > lim_up: break
            for k in range(reps[2]):
                mults[2] = k
                if sum(mults[u] * coins[u] for u in range(3)) > lim_up: break
                for l in range(reps[3]):
                    mults[3] = l
                    if sum(mults[u] * coins[u] for u in range(4)) > lim_up: break
                    for m in range(reps[4]):
                        mults[4] = m
                        if sum(mults[u] * coins[u] for u in range(5)) > lim_up: break
                        for n in range(reps[5]):
                            mults[5] = n
                            if sum(mults[u] * coins[u] for u in range(6)) > lim_up: break
                            for o in range(reps[6]):
                                mults[6] = o
                                if sum(mults[u] * coins[u] for u in range(7)) > lim_up: break
                                if check(mults):
                                    s = s + 1
    print(s)

Strategic


def count(size, remaining):
    if remaining == 0: return 1
    if remaining < 0 or ((size <= 0) and remaining >= 1): return 0
    return count(size - 1, remaining) + count(size, remaining - coins[size - 1])


def strategic():
    print(count(len(coins), lim_up))
