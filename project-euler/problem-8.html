def brute():

    seq = ""

Strategic solution


def strategic():
    splits = series.split("0")
    max = 0
    seq = ""
    for split in splits:
        if len(split) > lim_up:
            for i in range(len(series) - lim_up):
                a = 1
                for b in series[i:i + lim_up]: a = a * int(b)
                if a > max:
                    max = a
                    seq = "".join(series[i:i + lim_up])
    print(seq)
    print(max)
