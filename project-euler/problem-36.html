Double-base palindromes

Solved on Sun, 20 Dec 2020
Uploaded on Sun, 20 Dec 2020

The decimal number, 585 = 10010010012 (binary), is palindromic in both bases.

Find the sum of all numbers, less than one million, which are palindromic in base 10 and base 2.

(Please note that the palindromic number, in either base, may not include leading zeros.)

Brute-force solution


def brute():
    result = 0
    odd_palindromes = get_odd_palindromes()
    for palindrome in odd_palindromes:
        bin_palindrome = get_binary(palindrome)
        if bin_palindrome == bin_palindrome[::-1]:
            result = result + palindrome
    print(result)



def get_odd_palindromes():
    odd_palindromes = []
    for i in range(1, lim_up):
        if str(i) == str(i)[::-1]:
            if i % 2 != 0:
                odd_palindromes.append(i)
    return odd_palindromes


def get_binary(num):
    binary = ""
    power = 0
    while num >= 2**power:
        power = power + 1
    for p in list(range(power))[::-1]:
        if num == 0 or num < (2**p):
            binary = binary + "0"
        else:
            binary = binary + "1"
            num = num - (2**p)
    return str(binary)
