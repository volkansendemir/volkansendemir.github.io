215 = 32768 and the sum of its digits is 3 + 2 + 7 + 6 + 8 = 26.

What is the sum of the digits of the number 21000?

Brute-force solution


def get_sum_digits(num):
    sum_digits = 0
    for digit in str(num):
        sum_digits = sum_digits + int(digit)
    return sum_digits

Modular Visualization


from matplotlib import pyplot as plt


def visualize_remainders():
    digit_sums = []
    single_digit_sums = []

    for i in range(51):
        two_to_power = 2 ** i
        sum_digits = 0
        for digit in str(two_to_power):
            sum_digits = sum_digits + int(digit)
        digit_sums.append(sum_digits)
        single_digit_sums.append(sum_digits)
        while single_digit_sums[-1] > 9:
            sum_digits = 0
            for digit in str(single_digit_sums[-1]):
                sum_digits = sum_digits + int(digit)
            single_digit_sums[-1] = sum_digits

    plt.scatter(list(range(len(single_digit_sums))), single_digit_sums)
    plt.show()

    print(single_digit_sums[:21])
    print(digit_sums[:21])

Trend Visualization


def single_digit_sum(given_power):
    modular_sum = [1, 2, 4, 8, 7, 5]
    return modular_sum[given_power % 6]


def visualize_trend():
    digit_sums = []
    trend = []
    for i in list(range(1001))[single_digit_sum(1001)::6]:
        trend.append((9 * (i // 6)) + single_digit_sum(i))
        powered = 2 ** i
        powered_sum = 0
        for digit in str(powered):
            powered_sum = powered_sum + int(digit)
        digit_sums.append(powered_sum)

    plt.scatter(list(range(len(digit_sums))), digit_sums, s=5)
    plt.scatter(list(range(len(trend))), trend, s=3)
    plt.show()
