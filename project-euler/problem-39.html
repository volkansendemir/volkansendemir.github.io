If p is the perimeter of a right angle triangle with integral length sides, {a,b,c}, there are exactly three solutions for p = 120.

{20,48,52}, {24,45,51}, {30,40,50}

For which value of p â‰¤ 1000, is the number of solutions maximised?

Brute-force solution


def brute():
    max_reps = 0
    max_at = 0
    for i in range(1, lim_up):
        reps = 0
        for j in range(1, i):
            for k in range(1, min(j, i - j)):
                if ((i**2) - (2 * i * j) - (2 * i * k) + (2 * j * k)) == 0:
                    reps = reps + 1
                    print(i)
        if reps > max_reps:
            max_reps = reps
            max_at = i
    print(max_at, max_reps)

Strategic solution


def strategic():
    hits = [0 for i in range(lim_up)]
    for i in range(1, lim_up):
        for j in range(1, i):
            if i + j > lim_up: break
            c = ((i**2) + (j**2))**(1 / 2)
            if i + j + c > lim_up: break
            if int(c) == c: hits[int(i + j + c)] = hits[int(i + j + c)] + 1
    max_hit = max(hits)
    max_at = 0
    for i in range(lim_up):
        if hits[i] == max_hit:
            max_at = i
            break
    print(max_at, max_hit)
