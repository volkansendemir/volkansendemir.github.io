The sum of the primes below 10 is 2 + 3 + 5 + 7 = 17.

Find the sum of all the primes below two million.

Strategic solution


def strategic():
    primes = [2]
    integer_check = 3
    while primes[-1] < lim_up and integer_check <= lim_up:
        isPrime = True
        for prime in primes:
            if integer_check % prime == 0: isPrime = False
            if prime > integer_check ** (1/2): break
        if isPrime: primes.append(integer_check)
        integer_check = integer_check + 2
    print(sum(primes))

Brute-force solution


def brute():
    primes = [2]
    integer_check = 3
    while primes[-1] < lim_up:
        isPrime = True
        for prime in primes:
            if integer_check % prime == 0: isPrime = False
        if isPrime: primes.append(integer_check)
        integer_check = integer_check + 2
    print(sum(primes[:-1]))
    print(primes[:-1])
